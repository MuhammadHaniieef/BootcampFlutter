# Tambahkan Roaming\npm ke PATH User
$npmPath  = "$env:APPDATA\npm"
$userPath = [Environment]::GetEnvironmentVariable("Path","User")
if ($userPath -notlike "*$npmPath*") {
  [Environment]::SetEnvironmentVariable("Path", "$userPath;$npmPath", "User")
}

# Tambahkan Pub\Cache\bin ke PATH User
$pubBin   = "$env:LOCALAPPDATA\Pub\Cache\bin"
$userPath = [Environment]::GetEnvironmentVariable("Path","User")
if ($userPath -notlike "*$pubBin*") {
  [Environment]::SetEnvironmentVariable("Path", "$userPath;$pubBin", "User")
}
